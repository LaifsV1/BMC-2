ackermann (p : Int * Int) :(Int) =
    let (m:Int) = (fst:Int*Int) p in
    let (n:Int) = (snd:Int*Int) p in
    if m
    then
      if n
      then ackermann (m + -1,(ackermann (m,n + -1)))
      else ackermann (m + -1,1)
    else n+1;
